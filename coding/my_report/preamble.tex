%-------------------------------- Headers / footers ---------------------------
% DTU template: no running header; footer carries thesis short title + page
% number.  On even (left) pages: page# at left, title at right.
% On odd (right) pages: title at left, page# at right.
\usepackage{fancyhdr}
\pagestyle{fancy}
\fancyhf{}
\fancyfoot[LE]{\small\thepage}
\fancyfoot[RE]{\small\nouppercase{\thesisrunningtitle}}
\fancyfoot[LO]{\small\nouppercase{\thesisrunningtitle}}
\fancyfoot[RO]{\small\thepage}
\renewcommand{\headrulewidth}{0pt}
\renewcommand{\footrulewidth}{0pt}
% Same footer on plain pages (chapter opening pages)
\fancypagestyle{plain}{%
  \fancyhf{}%
  \fancyfoot[LE]{\small\thepage}%
  \fancyfoot[RE]{\small\nouppercase{\thesisrunningtitle}}%
  \fancyfoot[LO]{\small\nouppercase{\thesisrunningtitle}}%
  \fancyfoot[RO]{\small\thepage}%
  \renewcommand{\headrulewidth}{0pt}%
  \renewcommand{\footrulewidth}{0pt}%
}
